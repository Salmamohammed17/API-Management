package trial.sample;
import com.sun.jersey.api.client.Client;
import com.sun.jersey.api.client.ClientResponse;
import com.sun.jersey.api.client.WebResource;

import Decoder.BASE64Encoder;

public class Jerseysample {
	public static void main(String[] args) {
			try {
		        String user = "Administrator";
		        String pass = "manage";
		        String authString = user+ ":" + pass;
		        String authStringEnc = new BASE64Encoder().encode(authString.getBytes());
		        System.out.println("Base64 encoded auth string: " + authStringEnc);
		      Client client = Client.create();

				WebResource webResource = client
						.resource("http://localhost:5555/rest/apigateway/apis");
				  webResource.setProperty("Authorization", "Basic " + authStringEnc);

				ClientResponse response = webResource.accept("application/json").header("Authorization", "Basic " + authStringEnc)
						.get(ClientResponse.class);
			
		        

				if (response.getStatus() != 200) {
					throw new RuntimeException("Failed : HTTP error code : "
							+ response.getStatus());
				}

				String output = response.getEntity(String.class);

				System.out.println(output);

			} catch (Exception e) {

				e.printStackTrace();

			}

		}

	}
